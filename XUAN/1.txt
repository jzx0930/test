        public void btn_Connect_Click(object sender, EventArgs e)

        #region Update GUI Control

        private delegate void dEnableGroupControl(bool enable);
        private void enableGroupControls(bool enable)
        {
            if (group_Comm.InvokeRequired)
            {
                var func = new dEnableGroupControl(enableGroupControls);
                this.Invoke(func, enable);
            }
            else
            {
                group_Comm.Enabled = enable;
                group_JobOp.Enabled = enable;
                group_PLCStates.Enabled = enable;
            }
        }

        private delegate void dUpdatePnlBackColor(Panel panel, bool enable);
        private void updatePnlBackColor(Panel panel, bool enable)
        {
            if (panel.InvokeRequired)
            {
                var func = new dUpdatePnlBackColor(updatePnlBackColor);
                this.Invoke(func, panel, enable);
            }
            else
            {
                if (enable)
                    panel.BackColor = Color.LightGreen;
                else
                    panel.BackColor = Color.DarkGreen;
            }
        }

        private delegate void dUpdate(MainState state);
        private void updateMainStatusImage(MainState state)
        {
            string path = Environment.CurrentDirectory + "\\images\\" + state + ".png";
            Image temp = Image.FromFile(path);
            Bitmap bitmap = Module_Interface.ImageHandler.ResizeImage(temp, pnl_Main_Status.Width, pnl_Main_Status.Height);
            pnl_Main_Status.BackgroundImage = bitmap;
        }

        private void updateControlTextWithString(Control control, string text)
        {
            if (control.InvokeRequired)
            {
                Action<Control, string> del = (ctrl, msg) => ctrl.Text = msg;
                this.Invoke(del, control, text);
            }
            else
            {
                control.Text = text;
            }
        }

        #endregion

        #region BackgroundWorker

        private void BW_Connect_DoWork(object sender, DoWorkEventArgs e)
        {
            e.Result = e.Argument;
            MainState state = (MainState)e.Argument;
            if (state == MainState.Connect)
                DoConnect();
            else if (state == MainState.Disconnect)
                DoDisconnect();
            else
                throw new Exception("Undefined state = " + state);

            // Add delay
            Thread.Sleep(100);
        }

        private void BW_Connect_RunWorkerCompleted(object sender, RunWorkerCompletedEventArgs e)
        {
            if (e.Error != null)
                updateMainStatusImage(MainState.NO);
            else
            {
                updateMainStatusImage((MainState)e.Result);

                // Switch Group Controls
                if ((MainState)e.Result == MainState.Connect)
                {
                    enableGroupControls(true);
                    registerEventPLCState();
                }
                else if ((MainState)e.Result == MainState.Disconnect)
                    enableGroupControls(false);
            }
        }

        #region Event handling

        private void registerEventPLCState()
        {
            // Register the event
            _laserScanner.ArgesPLCEventUpdated += _argesScanner_ArgesPLCEventUpdated;

            void _argesScanner_ArgesPLCEventUpdated(object sender, ArgesPLCEventArgs e)
            {
                var states = e.States;
                // Update variable
                this.states = states;

                // Update UI controls
                updatePnlBackColor(pnl_DeviceAwake, states.PLC_DEVICES_AWAKE);
                updatePnlBackColor(pnl_DeviceFailure, states.PLC_DEVICES_FAILURE);
                updatePnlBackColor(pnl_DeviceReady, states.PLC_DEVICES_READY);
                updatePnlBackColor(pnl_DeviceSetup, states.PLC_DEVICES_SETUP);
                updatePnlBackColor(pnl_JobReady, states.PLC_JOB_READY);
                updatePnlBackColor(pnl_JobActive, states.PLC_JOB_ACTIVE);
                updatePnlBackColor(pnl_JobCompleted, states.PLC_JOB_COMPLETED);
                updatePnlBackColor(pnl_JobFailed, states.PLC_JOB_FAILED);
                updatePnlBackColor(pnl_SysSafe, states.PLC_SYS_SAFE);
                updatePnlBackColor(pnl_JobPaused, states.PLC_JOB_PAUSED);
                updatePnlBackColor(pnl_JobPiloted, states.PLC_JOB_PILOTED);
                updatePnlBackColor(pnl_JobPreloaded, states.PLC_JOB_PRELOADED);
                updatePnlBackColor(pnl_JobStopping, states.PLC_JOB_STOPPING);
                updatePnlBackColor(pnl_SystemReadyForPowerOff, states.PLC_READY_FOR_POWER_OFF);
                updatePnlBackColor(pnl_SystemReady, states.PLC_SYSTEM_READY);
                updatePnlBackColor(pnl_Attention, states.PLC_ATTENTION);
                updatePnlBackColor(pnl_SysSafeRequest, states.PLC_SYS_SAFE_REQUEST);
                updatePnlBackColor(pnl_JobPreload, states.PLC_JOB_PRELOAD);
            }
        }
